   ```python
   import os
   from pathlib import Path
   from convex import ConvexClient
   from dotenv import load_dotenv

   class PipelineConvexClient:
       """Convex client with admin auth for pipeline operations."""

       def __init__(self, convex_url: str | None = None, deploy_key: str | None = None):
           repo_root = Path(__file__).resolve().parent.parent.parent.parent.parent
           env_file = repo_root / ".env.local"
           if env_file.exists():
               load_dotenv(env_file)

           url = convex_url or os.getenv("CONVEX_URL")
           key = deploy_key or os.getenv("CONVEX_DEPLOY_KEY")
           if not url:
               raise ValueError("CONVEX_URL not set")
           if not key:
               raise ValueError("CONVEX_DEPLOY_KEY not set")

           self._client = ConvexClient(url)
           self._client.set_admin_auth(key)

       def mutation(self, name: str, args: dict) -> Any:
           return self._client.mutation(name, args)

       def query(self, name: str, args: dict | None = None) -> Any:
           return self._client.query(name, args or {})
   ```

5. **Create dedup.py** -- Within-source deduplication:
   ```python
   class SourceDeduplicator:
       """Track seen records within a single source to prevent re-yielding."""
       def __init__(self):
           self._seen_external_ids: set[str] = set()
           self._seen_source_urls: set[str] = set()

       def is_duplicate(self, record: dict, source_id: str) -> bool:
           """Check if record already seen in this scrape session."""
           ext_id = record.get("external_id")
           src_url = record.get("source_url", "")
           key_ext = f"{source_id}:{ext_id}" if ext_id else None
           key_url = f"{source_id}:{src_url}" if src_url else None
           if key_ext and key_ext in self._seen_external_ids:
               return True
           if key_url and key_url in self._seen_source_urls:
               return True
           if key_ext:
               self._seen_external_ids.add(key_ext)
           if key_url:
               self._seen_source_urls.add(key_url)
           return False
   ```

6. **Create differ.py** -- Field-level diff for change tracking:
   ```python
   DIFF_FIELDS = ["title", "description", "host_country", "application_deadline",
                  "award_amount", "source_url", "provider_organization", "application_url"]

   def compute_diff(old: dict, new: dict) -> list[tuple[str, str | None, str | None]]:
       """Return list of (field_name, old_value, new_value) for changed fields."""
       changes = []
       for field in DIFF_FIELDS:
           old_val = str(old.get(field, "")) if old.get(field) is not None else None
           new_val = str(new.get(field, "")) if new.get(field) is not None else None
           if old_val != new_val:
               changes.append((field, old_val, new_val))
       return changes
   ```

2. **Create retry.py** -- Exponential backoff retry decorator:
   ```python
   import asyncio
   import functools
   import structlog
   from typing import TypeVar, Callable

   logger = structlog.get_logger()

   RETRYABLE_STATUS_CODES = {429, 500, 502, 503, 504}

   async def retry_with_backoff(
       func: Callable,
       max_retries: int = 3,
       base_delay: float = 1.0,
       max_delay: float = 30.0,
       retryable_exceptions: tuple = (Exception,),
   ) -> Any:
       """Retry async function with exponential backoff."""
       for attempt in range(max_retries + 1):
           try:
               return await func()
           except retryable_exceptions as e:
               if attempt == max_retries:
                   raise
               delay = min(base_delay * (2 ** attempt), max_delay)
               # Add jitter: +/- 25%
               import random
               delay = delay * (0.75 + random.random() * 0.5)
               logger.warning("retry_attempt", attempt=attempt + 1, delay=delay, error=str(e))
               await asyncio.sleep(delay)
   ```
