Route definition:
```typescript
import { createFileRoute } from "@tanstack/react-router";
import { z } from "zod";

const compareSearchSchema = z.object({
  s: z.string().optional(), // comma-separated slugs
});

export const Route = createFileRoute("/scholarships/compare")({
  validateSearch: compareSearchSchema,
  head: ({ search }) => {
    // Dynamic title and OG tags based on scholarship names
    const slugs = search.s?.split(",").filter(Boolean) ?? [];
    // Title format per D-63: "{A} vs {B} Scholarship Comparison | ScholarHub"
    // Names will be populated client-side; fallback title for SSR
    return {
      meta: [
        { title: slugs.length > 0 ? "Scholarship Comparison | ScholarHub" : "Compare Scholarships | ScholarHub" },
      ],
    };
  },
  component: ComparePage,
});
```
