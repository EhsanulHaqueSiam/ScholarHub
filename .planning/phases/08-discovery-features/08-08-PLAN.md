```typescript
import { cronJobs } from "convex/server";
import { internal } from "./_generated/api";

const crons = cronJobs();

// Recompute collection scholarship counts (eventual consistency, D-90)
crons.daily("recompute collection counts", { hourUTC: 3, minuteUTC: 0 }, internal.collections.recomputeAllCounts);

// Refresh reverse related_ids (Pitfall 1: reverse updates via daily cron)
crons.daily("refresh related ids", { hourUTC: 4, minuteUTC: 0 }, internal.related.refreshAllRelatedIds);

```

Add corresponding internal mutations in collections.ts and related.ts:
- `recomputeAllCounts`: internal mutation that queries all active collections, for each runs the filter query and counts results, patches scholarship_count. Processes in batches of 10 collections with scheduler for remaining.
- `refreshAllRelatedIds`: internal mutation that queries published scholarships in batches of 50, recomputes related_ids for each, patches if changed. Uses scheduler for next batch.
