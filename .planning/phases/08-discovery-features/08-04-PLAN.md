**1. Create `web/src/components/comparison/CompareContext.tsx` (D-56):**

```typescript
import { createContext, useCallback, useContext, useState, type ReactNode } from "react";

interface CompareItem {
  slug: string;
  title: string;
}

interface CompareState {
  selected: CompareItem[];
  addToCompare: (slug: string, title: string) => void;
  removeFromCompare: (slug: string) => void;
  clearCompare: () => void;
  isSelected: (slug: string) => boolean;
  isFull: boolean;
}

const CompareContext = createContext<CompareState | null>(null);

export function CompareProvider({ children }: { children: ReactNode }) {
  const [selected, setSelected] = useState<CompareItem[]>([]);

  const addToCompare = useCallback((slug: string, title: string) => {
    setSelected((prev) => {
      if (prev.length >= 3) return prev;
      if (prev.some((item) => item.slug === slug)) return prev;
      return [...prev, { slug, title }];
    });
  }, []);

  const removeFromCompare = useCallback((slug: string) => {
    setSelected((prev) => prev.filter((item) => item.slug !== slug));
  }, []);

  const clearCompare = useCallback(() => setSelected([]), []);

  const isSelected = useCallback(
    (slug: string) => selected.some((item) => item.slug === slug),
    [selected],
  );

  return (
    <CompareContext.Provider
      value={{
        selected,
        addToCompare,
        removeFromCompare,
        clearCompare,
        isSelected,
        isFull: selected.length >= 3,
      }}
    >
      {children}
    </CompareContext.Provider>
  );
}

export function useCompare(): CompareState {
  const ctx = useContext(CompareContext);
  if (!ctx) throw new Error("useCompare must be used within CompareProvider");
  return ctx;
}
```

Session-only state -- cleared on refresh. No localStorage persistence per D-56.

**2. Create `web/src/components/comparison/CompareCheckbox.tsx` (D-54, D-99):**

Props: `{ slug: string; title: string; variant: "card" | "detail" | "listItem" }`.

Uses `useCompare()` context.

For `variant="card"`:
- Renders a checkbox in the top-right corner of the card
- Desktop: hidden by default, appears on parent hover (use group/group-hover)
- Mobile: always visible (no hover on touch devices)
- When checked: filled accent checkbox with checkmark icon (Check from lucide-react)
- When disabled (isFull && !isSelected): greyed out with Radix Tooltip "Maximum 3 scholarships can be compared at once"
- Keyboard: tabIndex={0}, onKeyDown handles Space/Enter to toggle
- ARIA: role="checkbox", aria-checked={isSelected}, aria-label={title}

```tsx
<div
  className={cn(
    "absolute top-2 right-2 z-10",
    variant === "card" && "opacity-0 group-hover:opacity-100 md:opacity-0 max-md:opacity-100 transition-opacity",
  )}
>
  <button
    type="button"
    role="checkbox"
    aria-checked={checked}
    aria-label={`Compare ${title}`}
    disabled={isFull && !checked}
    onClick={(e) => { e.preventDefault(); e.stopPropagation(); toggle(); }}
    onKeyDown={(e) => { if (e.key === " " || e.key === "Enter") { e.preventDefault(); toggle(); } }}
    className={cn(
      "size-6 rounded-base border-2 flex items-center justify-center transition-colors",
      checked
        ? "bg-main border-main text-main-foreground"
        : "bg-secondary-background border-border hover:border-main",
      isFull && !checked && "opacity-50 cursor-not-allowed",
    )}
  >
    {checked && <Check className="size-4" />}
  </button>
</div>
```

For `variant="detail"`:
- Full button: "Add to Compare" text with compare icon, or "Added to Compare" when selected
- Same functionality, different visual treatment

For `variant="listItem"`:
- Inline checkbox left of content, smaller (size-5)
