1. **Create .github/workflows/scrape.yml** -- Scheduled scraping workflow:
   ```yaml
   name: Scrape Scholarships

   on:
     schedule:
       - cron: '0 4 * * *'  # Daily at 04:00 UTC
     workflow_dispatch:
       inputs:
         source:
           description: 'Specific source name (blank for all due sources)'
           required: false
           type: string
         wave:
           description: 'Specific wave number (blank for all)'
           required: false
           type: string
         dry_run:
           description: 'Dry run (write to local JSON, skip Convex)'
           required: false
           type: boolean
           default: false

   permissions:
     contents: write      # For config auto-update commits
     issues: write        # For scraper rot GitHub Issues
     actions: read

   jobs:
     test:
       runs-on: ubuntu-latest
       timeout-minutes: 10
       steps:
         - uses: actions/checkout@v4

         - uses: astral-sh/setup-uv@v6

         - name: Install Python dependencies
           run: cd scraping && uv sync

         - name: Run tests (no browser-dependent tests)
           run: cd scraping && uv run pytest tests/ -x --timeout=30 -m "not browser"

     scrape:
       needs: test
       runs-on: ubuntu-latest
       timeout-minutes: 240  # 4-hour limit for full scrape
       steps:
         - uses: actions/checkout@v4

         - uses: astral-sh/setup-uv@v6

         - name: Install Python dependencies
           run: cd scraping && uv sync

         - name: Install Scrapling browser dependencies
           run: cd scraping && uv run scrapling install

         - name: Run scraping pipeline
           env:
             CONVEX_URL: ${{ secrets.CONVEX_URL }}
             CONVEX_DEPLOY_KEY: ${{ secrets.CONVEX_DEPLOY_KEY }}
             GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
             GITHUB_REPOSITORY: ${{ github.repository }}
           run: |
             ARGS="--json-logs"
             if [ "${{ inputs.source }}" != "" ]; then
               ARGS="$ARGS --source ${{ inputs.source }}"
             fi
             if [ "${{ inputs.wave }}" != "" ]; then
               ARGS="$ARGS --wave ${{ inputs.wave }}"
             fi
             if [ "${{ inputs.dry_run }}" == "true" ]; then
               ARGS="$ARGS --dry-run"
             fi
             cd scraping && uv run scrape run $ARGS

         - name: Upload buffer on failure
           if: failure()
           uses: actions/upload-artifact@v4
           with:
             name: scrape-buffer-${{ github.run_id }}
             path: scraping/.buffer/
             retention-days: 7

         - name: Upload run logs
           if: always()
           uses: actions/upload-artifact@v4
           with:
             name: scrape-logs-${{ github.run_id }}
             path: scraping/.logs/
             retention-days: 30
             if-no-files-found: ignore
   ```

1. **Create scraping/scripts/seed_test_data.py** -- Generate fake scholarship data in Convex:
   ```python
   """Seed Convex with realistic fake scholarship data for Phase 4/5 development."""
   import argparse
   import random
   import time
   from scholarhub_pipeline.ingestion.convex_client import PipelineConvexClient

   SAMPLE_TITLES = [
       "DAAD Research Grant for Doctoral Candidates",
       "Chevening Scholarship 2026",
       "Erasmus Mundus Joint Master Degrees",
       "MEXT Research Student Scholarship",
       "Fulbright Foreign Student Program",
       "Commonwealth Scholarships 2026",
       "Gates Cambridge Scholarship",
       "Rhodes Scholarship",
       "Swiss Government Excellence Scholarships",
       "China Scholarship Council Awards",
       # ... 40+ more realistic titles
   ]

   COUNTRIES = ["DE", "GB", "JP", "US", "AU", "FR", "NL", "SE", "NO", "CA", "CH", "AT", "KR", "CN", "SG"]
   DEGREE_LEVELS = ["bachelor", "master", "phd", "postdoc"]
   FUNDING_TYPES = ["fully_funded", "partial", "tuition_waiver", "stipend_only"]

   def generate_fake_scholarship(index: int) -> dict:
       """Generate a single fake scholarship record."""
       return {
           "title": f"{random.choice(SAMPLE_TITLES)} #{index}",
           "description": f"A scholarship opportunity for international students. Application #{index}.",
           "provider_organization": random.choice(["DAAD", "British Council", "JASSO", "IIE", "CSC"]),
           "host_country": random.choice(COUNTRIES),
           "degree_levels": random.sample(DEGREE_LEVELS, k=random.randint(1, 3)),
           "funding_type": random.choice(FUNDING_TYPES),
           "award_amount": str(random.randint(5000, 50000)),
           "award_currency": random.choice(["EUR", "GBP", "USD", "JPY"]),
           "application_deadline": f"2026-{random.randint(1,12):02d}-{random.randint(1,28):02d}",
           "source_url": f"https://example.com/scholarship/{index}",
           "scraped_at": int(time.time() * 1000),
           "quality_flags": [],
       }

   def main():
       parser = argparse.ArgumentParser(description="Seed Convex with fake scholarship data")
       parser.add_argument("--count", type=int, default=100, help="Number of records to generate")
       parser.add_argument("--dry-run", action="store_true", help="Print data without inserting")
       args = parser.parse_args()

       if args.dry_run:
           for i in range(min(5, args.count)):
               print(json.dumps(generate_fake_scholarship(i), indent=2))
           return

       client = PipelineConvexClient()
       # Create a test run
       run_id = client.mutation("scraping:startRun", {
           "triggered_by": "seed_script",
           "sources_targeted": 1,
       })
       # Batch insert
       batch = []
       for i in range(args.count):
           batch.append(generate_fake_scholarship(i))
           if len(batch) >= 50:
               client.mutation("scraping:batchInsertRawRecords", {"records": batch, "run_id": run_id})
               batch = []
       if batch:
           client.mutation("scraping:batchInsertRawRecords", {"records": batch, "run_id": run_id})
       # Complete run
       client.mutation("scraping:completeRun", {
           "run_id": run_id, "status": "completed",
           "sources_completed": 1, "sources_failed": 0,
           "records_inserted": args.count, "records_updated": 0, "records_unchanged": 0,
           "duration_seconds": 1,
       })
       print(f"Seeded {args.count} fake scholarships")

   if __name__ == "__main__":
       main()
   ```
