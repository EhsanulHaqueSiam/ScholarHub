1. **Create scheduler.py** -- Source filtering and grouping:
   ```python
   import time
   import structlog
   from scholarhub_pipeline.configs._protocol import SourceConfig

   logger = structlog.get_logger()

   METHOD_GROUPS = ["api", "jsonld", "ajax", "rss", "scrape", "scrapling"]

   class SourceScheduler:
       """Filter and group sources for execution."""

       def __init__(self, convex_client):
           self.convex = convex_client

       def filter_due_sources(self, configs: list[SourceConfig]) -> list[SourceConfig]:
           """Return only configs whose frequency has elapsed since last scrape."""
           due = []
           for config in configs:
               # Query sources table for last_scraped
               source = self.convex.query("sources:getByName", {"name": config.name})
               if source is None:
                   due.append(config)  # Never scraped
                   continue
               last_scraped = source.get("last_scraped")
               if last_scraped is None:
                   due.append(config)
                   continue
               hours_since = (time.time() * 1000 - last_scraped) / (1000 * 60 * 60)
               freq = getattr(config, 'scrape_frequency_hours', 168)
               if hours_since >= freq:
                   due.append(config)
               else:
                   logger.debug("source_not_due", name=config.name, hours_since=round(hours_since, 1))
           return due

       def group_by_method(self, configs: list[SourceConfig]) -> dict[str, list[SourceConfig]]:
           """Group configs by primary_method for execution grouping."""
           groups: dict[str, list[SourceConfig]] = {}
           for config in configs:
               method = config.primary_method
               if method not in groups:
                   groups[method] = []
               groups[method].append(config)
           return groups

       def filter_active(self, configs: list[SourceConfig]) -> list[SourceConfig]:
           """Exclude auth_required and deactivated sources."""
           return [c for c in configs if not getattr(c, 'auth_config', None)]
   ```

2. **Create buffer.py** -- Local JSON buffer for Convex downtime:
   ```python
   import json
   import structlog
   from pathlib import Path
   from datetime import datetime, timezone

   logger = structlog.get_logger()

   class LocalBuffer:
       """Buffer records to local JSON files when Convex is unreachable."""

       def __init__(self, buffer_dir: str = ".buffer"):
           self.buffer_dir = Path(buffer_dir)
           self.buffer_dir.mkdir(parents=True, exist_ok=True)

       def save(self, records: list[dict], source_name: str) -> Path:
           """Save records to a JSON file."""
           ts = datetime.now(tz=timezone.utc).strftime("%Y%m%dT%H%M%S")
           filename = f"{source_name}_{ts}.json"
           filepath = self.buffer_dir / filename
           filepath.write_text(json.dumps(records, default=str, indent=2))
           logger.info("records_buffered", file=str(filepath), count=len(records))
           return filepath

       def load_all(self) -> list[tuple[Path, list[dict]]]:
           """Load all buffered files, return list of (path, records)."""
           results = []
           for f in sorted(self.buffer_dir.glob("*.json")):
               data = json.loads(f.read_text())
               results.append((f, data))
           return results

       def clear(self, filepath: Path) -> None:
           """Remove a processed buffer file."""
           filepath.unlink(missing_ok=True)
   ```

3. **Create runner.py** -- Pipeline orchestrator:
   ```python
   import asyncio
   import time
   import structlog
   from scholarhub_pipeline.configs import discover_configs
   from scholarhub_pipeline.scrapers import get_scraper
   from scholarhub_pipeline.ingestion.convex_client import PipelineConvexClient
   from scholarhub_pipeline.ingestion.batch import BatchAccumulator
   from scholarhub_pipeline.ingestion.dedup import SourceDeduplicator
   from scholarhub_pipeline.monitoring.health import HealthTracker
   from scholarhub_pipeline.monitoring.rot_detector import RotDetector
   from scholarhub_pipeline.monitoring.github_issues import GitHubIssueManager
   from scholarhub_pipeline.monitoring.heartbeat import HeartbeatMonitor
   from scholarhub_pipeline.pipeline.scheduler import SourceScheduler
   from scholarhub_pipeline.pipeline.buffer import LocalBuffer

   logger = structlog.get_logger()

   class PipelineRunner:
       """Orchestrates a complete scrape run."""

       def __init__(
           self,
           convex_client: PipelineConvexClient | None = None,
           dry_run: bool = False,
           source_filter: str | None = None,
           wave_filter: int | None = None,
           json_logs: bool = False,
       ):
           self.dry_run = dry_run
           self.source_filter = source_filter
           self.wave_filter = wave_filter
           self.convex = convex_client
           self.buffer = LocalBuffer()
           self.stats = {
               "sources_targeted": 0, "sources_completed": 0, "sources_failed": 0,
               "records_inserted": 0, "records_updated": 0, "records_unchanged": 0,
           }

       async def run(self) -> dict:
           """Execute full pipeline run."""
           start_time = time.time()

           # Discover and filter configs
           all_configs = discover_configs()
           scheduler = SourceScheduler(self.convex) if self.convex else None

           configs = all_configs
           if self.source_filter:
               configs = [c for c in configs if c.name == self.source_filter or c.source_id == self.source_filter]
           if self.wave_filter:
               configs = [c for c in configs if getattr(c, 'wave', None) == self.wave_filter]
           if scheduler and not self.dry_run:
               configs = scheduler.filter_active(configs)
               configs = scheduler.filter_due_sources(configs)

           self.stats["sources_targeted"] = len(configs)
           logger.info("pipeline_start", sources=len(configs), dry_run=self.dry_run)

           # Start run in Convex
           run_id = None
           if self.convex and not self.dry_run:
               run_id = self.convex.mutation("scraping:startRun", {
                   "triggered_by": "cli",
                   "sources_targeted": len(configs),
               })

           # Group by method and execute
           groups = SourceScheduler.group_by_method(None, configs) if configs else {}

           # Execute in method order: api, jsonld, ajax, rss, scrape, scrapling
           for method in ["api", "jsonld", "ajax", "rss", "scrape", "scrapling"]:
               method_configs = groups.get(method, [])
               for config in method_configs:
                   await self._scrape_source(config, run_id)

           # Complete run
           duration = time.time() - start_time
           if self.convex and run_id and not self.dry_run:
               self.convex.mutation("scraping:completeRun", {
                   "run_id": run_id,
                   "status": "completed",
                   **self.stats,
                   "duration_seconds": int(duration),
               })
               HeartbeatMonitor(self.convex).update()

           logger.info("pipeline_complete", duration=round(duration, 1), **self.stats)
           return self.stats

       async def _scrape_source(self, config, run_id: str | None) -> None:
           """Scrape a single source, handle errors, update health."""
           source_start = time.time()
           try:
               scraper = get_scraper(config)
               records = await scraper.scrape()

               if self.dry_run:
                   self.buffer.save(records, config.source_id)
                   self.stats["records_inserted"] += len(records)
               elif self.convex and run_id:
                   batch = BatchAccumulator(self.convex, run_id)
                   dedup = SourceDeduplicator()
                   for record in records:
                       if not dedup.is_duplicate(record, config.source_id):
                           record["source_id"] = config.source_id
                           batch.add(record)
                   result = batch.flush_remaining()
                   self.stats["records_inserted"] += result.get("inserted", 0)
                   self.stats["records_updated"] += result.get("updated", 0)
                   self.stats["records_unchanged"] += result.get("unchanged", 0)

               duration = time.time() - source_start
               self.stats["sources_completed"] += 1

               # Update health and timestamp
               if self.convex and not self.dry_run:
                   HealthTracker(self.convex).record_success(config.source_id, len(records))
                   self.convex.mutation("scraping:updateLastScraped", {
                       "source_id": config.source_id,
                       "timestamp": int(time.time() * 1000),
                   })
                   self.convex.mutation("scraping:recordSourceResult", {
                       "run_id": run_id,
                       "source_id": config.source_id,
                       "status": "success",
                       "method_used": config.primary_method,
                       "records_found": scraper.records_found,
                       "records_new": len(records),
                       "records_updated": 0,
                       "records_unchanged": 0,
                       "duration_seconds": int(duration),
                       "bytes_downloaded": scraper.bytes_downloaded,
                   })

               logger.info("source_complete", source=config.name, records=len(records), duration=round(duration, 1))

           except Exception as e:
               duration = time.time() - source_start
               self.stats["sources_failed"] += 1
               error_type = RotDetector().classify_error(None, e)
               logger.error("source_failed", source=config.name, error=str(e), error_type=error_type)

               if self.convex and not self.dry_run:
                   health_result = HealthTracker(self.convex).record_failure(config.source_id, error_type, str(e))
                   # Check if we need to alert
                   rot = RotDetector()
                   failures = health_result.get("consecutive_failures", 0) if health_result else 0
                   if rot.should_alert(failures):
                       mgr = GitHubIssueManager(self.convex)
                       mgr.create_rot_issue(
                           source_name=config.name,
                           source_url=config.url,
                           error_type=error_type,
                           consecutive_failures=failures,
                           last_success=None,
                           suggested_fix=mgr.suggest_fix(error_type, config.url),
                       )
   ```
