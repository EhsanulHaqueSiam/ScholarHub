**4. Copy Link Button on Cards**

In ScholarshipCard and ScholarshipListItem, implement the "Copy Link" button:
- Button text: "Copy Link"
- On click: `navigator.clipboard.writeText(window.location.origin + "/scholarships/" + scholarship.slug)`
- After copy: button text changes to "Copied!" for 2 seconds, then reverts. Use `useState` with `setTimeout`.
- `aria-label="Copy link to ${scholarship.title}"`
- Stop propagation so clicking the button doesn't navigate to the detail page (since the entire card is a link)

```tsx
function CopyLinkButton({ slug, title }: { slug: string; title: string }) {
  const [copied, setCopied] = useState(false);
  const handleCopy = (e: React.MouseEvent) => {
    e.preventDefault();
    e.stopPropagation();
    navigator.clipboard.writeText(`${window.location.origin}/scholarships/${slug}`);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };
  return (
    <Button variant="neutral" size="sm" onClick={handleCopy} aria-label={`Copy link to ${title}`}>
      {copied ? "Copied!" : "Copy Link"}
    </Button>
  );
}
```

**5. Error Handling**

Create `web/src/components/ErrorBoundary.tsx` -- a class-based React error boundary:

```tsx
import { Component, type ReactNode } from "react";

interface Props {
  children: ReactNode;
  fallback: (props: { error: Error; reset: () => void }) => ReactNode;
}

interface State {
  error: Error | null;
}

export class ErrorBoundary extends Component<Props, State> {
  state: State = { error: null };

  static getDerivedStateFromError(error: Error): State {
    return { error };
  }

  reset = () => {
    this.setState({ error: null });
  };

  render() {
    if (this.state.error) {
      return this.props.fallback({ error: this.state.error, reset: this.reset });
    }
    return this.props.children;
  }
}
```

Create a `DirectoryErrorFallback` in the directory route:

```tsx
function DirectoryErrorFallback({ error, reset }: { error: Error; reset: () => void }) {
  return (
    <div className="border-2 border-[var(--urgency-critical)] rounded-base p-6 bg-[var(--urgency-critical)]/10 text-center">
      <p className="font-heading text-lg">Unable to load scholarships</p>
      <p className="mt-2 text-sm">Check your connection and try again.</p>
      <Button variant="default" className="mt-4" onClick={reset}>
        Retry
      </Button>
    </div>
  );
}
```

Wrap the results section in `<ErrorBoundary fallback={(props) => <DirectoryErrorFallback {...props} />}>`.
