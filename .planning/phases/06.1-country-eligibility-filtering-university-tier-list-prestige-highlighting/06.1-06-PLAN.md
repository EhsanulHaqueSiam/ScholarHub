1. **Create `web/src/routes/scholarships/index.tsx`** -- Main directory page:

This is the primary page of the application. Uses `createFileRoute("/scholarships/")`.

**Route definition:**
```typescript
import { createFileRoute } from "@tanstack/react-router";
import { scholarshipSearchSchema } from "@/lib/filters";

export const Route = createFileRoute("/scholarships/")({
  validateSearch: scholarshipSearchSchema,
  head: ({ search }) => {
    // Dynamic meta tags based on filter state
    const parts: string[] = [];
    if (search.degree) parts.push(formatDegree(search.degree));
    parts.push("Scholarships");
    if (search.to) parts.push(`in ${formatCountries(search.to)}`);
    if (search.from) parts.push(`for ${formatNationalities(search.from)} Students`);
    const title = parts.join(" ") + " — ScholarHub";
    const description = `Browse international scholarships${search.to ? ` in ${formatCountries(search.to)}` : ""}. Filter by degree, funding, and eligibility.`;

    return {
      meta: [
        { title },
        { name: "description", content: description },
        { property: "og:title", content: title },
        { property: "og:description", content: description },
        { property: "og:type", content: "website" },
      ],
    };
  },
  component: ScholarshipsDirectory,
});
```

Helper functions `formatDegree`, `formatCountries`, `formatNationalities` convert comma-separated codes to human-readable text (e.g., "BD,IN" -> "Bangladeshi, Indian"). Use `getCountryName` from countries.ts.

**Page layout (top to bottom):**

```tsx
function ScholarshipsDirectory() {
  const { filters, queryArgs, setFilter, clearFilters, removeFilter, activeFilterCount } = useScholarshipFilters();
  const { results, status, loadMore } = usePaginatedQuery(
    api.directory.listScholarships,
    { ...queryArgs, status: "published" },
    { initialNumItems: 20 }
  );
  const totalCount = useQuery(api.directory.getScholarshipCount, { status: "published" });

  return (
    <>
      <Navbar />
      {/* Hero Section */}
      <section className="w-full bg-background py-12 md:py-16 text-center">
        <h1 className="font-heading text-[32px] leading-[1.1]">
          Browse {totalCount ? `${totalCount.toLocaleString()}+` : "2,400+"} International Scholarships
        </h1>
        <p className="mt-4 text-base font-base max-w-2xl mx-auto">
          Find fully funded scholarships you qualify for. Filter by country, degree, and prestige.
        </p>
        <div className="mt-8 max-w-xl mx-auto px-4">
          <SearchBar onSearch={(q) => setFilter("q", q)} defaultValue={filters.q} />
        </div>
      </section>

      {/* Nationality Detection Banner */}
      <NationalityBanner />

      {/* Smart Eligibility Filter Bar */}
      <EligibilityFilterBar />

      {/* Directory Section */}
      <section className="max-w-[1280px] mx-auto px-4 py-8">
        {/* Featured Row */}
        <FeaturedRow nationalities={filters.from} />

        {/* Quick Filters */}
        <QuickFilters />

        {/* Active Filter Chips */}
        <FilterChips />

        {/* Sort + View + Count row */}
        <div className="flex items-center justify-between flex-wrap gap-4 mt-6 mb-4">
          <div className="flex items-center gap-2">
            <SortPills />
            <ViewToggle />
          </div>
          <p className="text-sm font-base font-semibold">
            {formatResultsCount(results.length, filters, totalCount)}
          </p>
        </div>

        {/* Content: Sidebar + Results */}
        <div className="flex gap-8">
          {/* Filter Panel - hidden on mobile, shown on desktop */}
          <div className="hidden lg:block w-[280px] shrink-0">
            <FilterPanel />
          </div>

          {/* Results Area */}
          <div className="flex-1 min-w-0">
            {/* Mobile Filter Button */}
            <div className="lg:hidden mb-4">
              <FilterPanel /> {/* Mobile mode auto-detected inside component */}
            </div>

            {status === "LoadingFirstPage" ? (
              <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 lg:gap-6">
                {Array.from({ length: 6 }).map((_, i) => <SkeletonCard key={i} />)}
              </div>
            ) : results.length === 0 ? (
              <EmptyState />
            ) : (
              <>
                {filters.view === "list" ? (
                  <div className="flex flex-col gap-4">
                    {results.map((s) => <ScholarshipListItem key={s._id} scholarship={s} />)}
                  </div>
                ) : (
                  <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 lg:gap-6">
                    {results.map((s) => <ScholarshipCard key={s._id} scholarship={s} />)}
                  </div>
                )}

                {/* Load More */}
                {status === "CanLoadMore" && (
                  <div className="mt-8 text-center">
                    <Button variant="neutral" onClick={() => loadMore(20)}>
                      Load More Scholarships
                    </Button>
                  </div>
                )}
                {status === "LoadingMore" && (
                  <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 lg:gap-6 mt-4">
                    {Array.from({ length: 3 }).map((_, i) => <SkeletonCard key={i} />)}
                  </div>
                )}
                {status === "Exhausted" && results.length > 0 && (
                  <p className="mt-8 text-center text-sm text-foreground/60">
                    You've seen all {results.length} matching scholarships
                  </p>
                )}
              </>
            )}

            {/* Error handling */}
            {/* Wrap the query usage in an ErrorBoundary or use Convex error handling */}
          </div>
        </div>
      </section>

      <BackToTop />
    </>
  );
}
```
